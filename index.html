        async function getRealStreetRoute(coords) {

            if (!coords || coords.length < 2) return { geometry: coords || [], distance: 0 };

            const points = coords.map(c => `${c[1]},${c[0]}`).join(';');

            const osrmUrl = `https://router.project-osrm.org/route/v1/driving/${points}?overview=full&geometries=geojson`;

            try {

                const res = await fetch(osrmUrl);

                const json = await res.json();

                if (json.code === 'Ok' && json.routes?.[0]) {

                    const streetGeometry = json.routes[0].geometry.coordinates.map(c => [c[1], c[0]]);

                    return { geometry: streetGeometry, distance: json.routes[0].distance / 1000 };

                }

            } catch (e) { console.error(e); }

            return { geometry: coords, distance: 0 };

        }


        async function updateMapRoute(data) {

            if (!map || !data.coords || data.coords.length < 1) return;

            if (polyline) map.removeLayer(polyline);

            markers.forEach(m => map.removeLayer(m));

            markers = [];


            const routeResult = await getRealStreetRoute(data.coords);

            window.tripDistances[data.id] = routeResult.distance;

            updateTotalKmDisplay();


            polyline = L.polyline(routeResult.geometry, { 

                color: '#10b981', weight: 6, opacity: 0.8, lineCap: 'round'

            }).addTo(map);


            data.coords.forEach((c, idx) => {

                const m = L.circleMarker(c, { radius: 7, color: '#000', fillColor: idx === 0 ? '#10b981' : '#f59e0b', fillOpacity: 1, weight: 2 }).addTo(map);

                markers.push(m);

            });


            if (routeResult.geometry.length > 0) map.fitBounds(polyline.getBounds(), { padding: [50, 50] });

            const kmText = routeResult.distance.toFixed(1);

            document.getElementById('map-km-label').innerText = `Percurso: ${kmText}km via ruas`;

            document.getElementById('map-day-km').innerText = kmText;

        }

        async function getRealStreetRoute(coords) {

            if (!coords || coords.length < 2) return { geometry: coords || [], distance: 0 };

            const points = coords.map(c => `${c[1]},${c[0]}`).join(';');

            const osrmUrl = `https://router.project-osrm.org/route/v1/driving/${points}?overview=full&geometries=geojson`;

            try {

                const res = await fetch(osrmUrl);

                const json = await res.json();

                if (json.code === 'Ok' && json.routes?.[0]) {

                    const streetGeometry = json.routes[0].geometry.coordinates.map(c => [c[1], c[0]]);

                    return { geometry: streetGeometry, distance: json.routes[0].distance / 1000 };

                }

            } catch (e) { console.error(e); }

            return { geometry: coords, distance: 0 };

        }


        async function updateMapRoute(data) {

            if (!map || !data.coords || data.coords.length < 1) return;

            if (polyline) map.removeLayer(polyline);

            markers.forEach(m => map.removeLayer(m));

            markers = [];


            const routeResult = await getRealStreetRoute(data.coords);

            window.tripDistances[data.id] = routeResult.distance;

            updateTotalKmDisplay();


            polyline = L.polyline(routeResult.geometry, { 

                color: '#10b981', weight: 6, opacity: 0.8, lineCap: 'round'

            }).addTo(map);


            data.coords.forEach((c, idx) => {

                const m = L.circleMarker(c, { radius: 7, color: '#000', fillColor: idx === 0 ? '#10b981' : '#f59e0b', fillOpacity: 1, weight: 2 }).addTo(map);

                markers.push(m);

            });


            if (routeResult.geometry.length > 0) map.fitBounds(polyline.getBounds(), { padding: [50, 50] });

            const kmText = routeResult.distance.toFixed(1);

            document.getElementById('map-km-label').innerText = `Percurso: ${kmText}km via ruas`;

            document.getElementById('map-day-km').innerText = kmText;

        }


        function updateTotalKmDisplay() {
